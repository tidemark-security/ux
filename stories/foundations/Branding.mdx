export const parseColorChannels = (color) => {
  const match = color.match(/rgba?\(([^)]+)\)/);

  if (!match) {
    return null;
  }

  const channels = match[1]
    .split(",")
    .map((channel) => Number(channel.trim().replace("/", "")));

  if (channels.length < 3 || channels.slice(0, 3).some((channel) => Number.isNaN(channel))) {
    return null;
  }

  return {
    red: Math.round(channels[0]),
    green: Math.round(channels[1]),
    blue: Math.round(channels[2]),
    alpha: Number.isNaN(channels[3]) || channels[3] === undefined ? 1 : channels[3],
  };
};

export const colorToHex = (color) => {
  const channels = parseColorChannels(color);

  if (!channels) {
    return "";
  }

  const toHex = (channel) => Math.max(0, Math.min(255, channel)).toString(16).padStart(2, "0");
  const alphaHex = channels.alpha < 1 ? toHex(Math.round(channels.alpha * 255)) : "";

  return `#${toHex(channels.red)}${toHex(channels.green)}${toHex(channels.blue)}${alphaHex}`;
};

export const normalizeRgb = (color) => {
  const channels = parseColorChannels(color);

  if (!channels) {
    return color;
  }

  if (channels.alpha < 1) {
    return `rgba(${channels.red}, ${channels.green}, ${channels.blue}, ${Number(channels.alpha.toFixed(3))})`;
  }

  return `rgb(${channels.red}, ${channels.green}, ${channels.blue})`;
};

export function CopyValue({ value }) {
  const [copied, setCopied] = useState(false);

  const handleCopy = async () => {
    if (!value) {
      return;
    }

    await navigator.clipboard.writeText(value);
    setCopied(true);
    window.setTimeout(() => setCopied(false), 900);
  };

  return (
    <button
      type="button"
      onClick={handleCopy}
      title={`Copy ${value}`}
      style={{
        display: "block",
        width: "100%",
        padding: "2px 0",
        color: copied ? "rgb(var(--color-focus-border))" : "rgb(var(--color-subtext-color))",
        fontFamily: '"Kode Mono", monospace',
        fontSize: "12px",
        lineHeight: "18px",
        textAlign: "left",
        wordBreak: "break-word",
        background: "transparent",
        border: 0,
        cursor: "copy",
      }}
    >
      {value}
    </button>
  );
}

export function ColorSwatch({ className, sample }) {
  const sampleRef = useRef(null);
  const [computedColor, setComputedColor] = useState(sample);

  useEffect(() => {
    const updateColor = () => {
      if (!sampleRef.current) {
        return;
      }

      setComputedColor(normalizeRgb(window.getComputedStyle(sampleRef.current).backgroundColor));
    };

    updateColor();

    const observer = new MutationObserver(updateColor);
    observer.observe(document.documentElement, { attributes: true });
    observer.observe(document.body, { attributes: true });

    return () => observer.disconnect();
  }, [sample]);

  const rgbValue = normalizeRgb(computedColor);
  const hexValue = colorToHex(rgbValue);

  return (
    <div className="ux-storybook-panel ux-storybook-panel-flat bevel-tr-3xl" style={{ overflow: "hidden" }}>
      <div ref={sampleRef} style={{ height: "72px", background: sample }} />
      <div style={{ padding: "12px" }}>
        <CopyValue value={className} />
        <CopyValue value={hexValue} />
        <CopyValue value={rgbValue} />
      </div>
    </div>
  );
}
